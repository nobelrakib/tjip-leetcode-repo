public class Solution {
    public bool IsAnagram(string s, string t) {
        if(s.Length != t.Length) return false;
        var dic1 = new Dictionary<char,int>();
        var dic2 = new Dictionary<char,int>();

        foreach(var c in s){
            if(!dic1.ContainsKey(c)) dic1.Add(c,1);
            else dic1[c] +=1;
        }
        foreach(var c in t){
            if(!dic2.ContainsKey(c)) dic2.Add(c,1);
            else dic2[c] +=1;
        }
        var isAnargam = true;
        foreach(var c in t){
            
            if(!dic1.ContainsKey(c) || dic1[c] != dic2[c]) {
                isAnargam=false;
                break;
            }
        }
        
        return isAnargam;
    }
}
